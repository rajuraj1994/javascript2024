onclick -> when a user click on the html element 
        ondblclick -> double click 
        onload -> when a page is fully loaded 
        onmouseover -> when mouse puts over the html element
        onmouseout -> when mouse puts away from that html elements after mouseover
        onchange ->  when a input field is changed 
        onkeyup -> when a key is released after press 
        onkeydown -> when a key is press down
        onsubmit -> when a form is submitted 
        onmousemove ->when mouse moves around the element
